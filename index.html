Menggunakan Named Constructor (.tanpaNama) print("\n4. Named Constructor (.tanpaNama):"); Mahasiswa mhs3 = Mahasiswa.tanpaNama(); mhs3.tampilkanData(); // 5. Generative & Default Constructor (Poin 1 & Latihan a) // Menggunakan Initializer List untuk mengatur status secara otomatis Mahasiswa(this.nama, this.nim, this._nilai) : status = (_nilai >= 70) ? "Lulus" : "Remedial"; // B. Named Constructor: Mahasiswa.dariMap (Latihan b) Mahasiswa.dariMap(Map map) : nama = map["nama"], nim = map["nim"], _nilai = map["nilai"].toDouble(), status = (map["nilai"] >= 70) ? "Lulus" : "Remedial"; // C. Named Constructor: Mahasiswa.tanpaNama (Latihan c) Mahasiswa.tanpaNama() : nama = "Belum Diisi", nim = "000000000", _nilai = 0.0, status = "Belum Diketahui"; // D. "Lulus" : "Remedial"; } else { print(" [Peringatan]: Nilai harus berada di antara 0 sampai 100!"); } } // Metode untuk mencetak data ke konsol void tampilkanData() { print(" Nama : $nama"); print(" NIM : $nim"); print(" Status : $status"); } } // F. Constant Constructor (Poin 2) // Semua variabel wajib bersifat 'final' dan ditambahkan kata kunci 'const' class KategoriMateri { final String namaMateri; final int sks; const KategoriMateri(this.namaMateri, this.sks); }
